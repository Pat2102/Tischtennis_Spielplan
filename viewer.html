// Rangliste basierend auf den Ergebnissen generieren get(ref(db, 'spielplan')).then(snapshot => { if (snapshot.exists()) { const spielplan = snapshot.val(); const teamStats = {}; spielplan.forEach(match => { const [team1, team2] = [match.team1, match.team2]; const results = match.result; // Wenn das Spiel noch nicht gespielt wurde, überspringen if (!results[0]) return; // Initialisiere Team-Statistiken, wenn noch nicht vorhanden if (!teamStats[team1]) { teamStats[team1] = { wins: 0, losses: 0, scored: 0, conceded: 0, games: 0 }; } if (!teamStats[team2]) { teamStats[team2] = { wins: 0, losses: 0, scored: 0, conceded: 0, games: 0 }; } // Berechnung der Ergebnisse für alle Sätze let team1Points = 0; let team2Points = 0; results.forEach(result => { const [score1, score2] = result.split('-').map(Number); team1Points += score1 || 0; team2Points += score2 || 0; }); // Berechnung der Gewinner if (team1Points > team2Points) { teamStats[team1].wins += 2; // 2 Punkte für den Sieg teamStats[team2].losses += 2; // 0 Punkte für die Niederlage } else if (team2Points > team1Points) { teamStats[team2].wins += 2; // 2 Punkte für den Sieg teamStats[team1].losses += 2; // 0 Punkte für die Niederlage } // Gesamtpunkte berechnen (erzielt und gegen erhaltene Punkte) teamStats[team1].scored += team1Points; teamStats[team2].scored += team2Points; teamStats[team1].conceded += team2Points; teamStats[team2].conceded += team1Points; // Anzahl der absolvierten Spiele erhöhen teamStats[team1].games += 1; teamStats[team2].games += 1; }); // Rangliste sortieren und anzeigen const sortedTeams = Object.entries(teamStats) .map(([team, stats]) => ({ team, points: `${stats.wins}:${stats.losses}`, games: stats.games, // Anzahl der Spiele anzeigen diff: stats.scored - stats.conceded, // Differenz berechnen scored: stats.scored, conceded: stats.conceded })) .sort((a, b) => { const [aWins] = a.points.split(':').map(Number); const [bWins] = b.points.split(':').map(Number); return bWins - aWins || b.diff - a.diff; }); sortedTeams.forEach((team, index) => { const row = document.createElement('tr'); row.innerHTML = ` ${index + 1} ${team.team} ${team.games} ${team.points} ${team.scored}:${team.conceded} ${team.diff} `; rankingTable.appendChild(row); }); } else { console.log('Keine Spielplan-Daten für Rangliste gefunden.'); } }); } // Beim Laden der Seite die Daten aus Firebase laden window.onload = loadTeamsAndSchedule; // Echtzeit-Updates für Änderungen an den Daten onChildChanged(ref(db, 'spielplan'), loadTeamsAndSchedule); onChildAdded(ref(db, 'spielplan'), loadTeamsAndSchedule);
